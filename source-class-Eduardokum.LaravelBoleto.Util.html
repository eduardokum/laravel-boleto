 323     /**
 324      * Mostra o Valor no float Formatado
 325      *
 326      * @param  string  $number
 327      * @param  integer $decimals
 328      * @param  boolean $showThousands
 329      * @return string
 330      */
 331     public static function nFloat($number, $decimals = 2, $showThousands = false)
 332     {
 333         if (is_null($number) || empty(self::onlyNumbers($number)) || floatval($number) == 0) {
 334             return 0;
 335         }
 336         $pontuacao = preg_replace('/[0-9]/', '', $number);
 337         $locale = (mb_substr($pontuacao, -1, 1) == ',') ? "pt-BR" : "en-US";
 338         $formater = new \NumberFormatter($locale, \NumberFormatter::DECIMAL);
 339 
 340         if ($decimals === false) {
 341             $decimals = 2;
 342             preg_match_all('/[0-9][^0-9]([0-9]+)/', $number, $matches);
 343             if (!empty($matches[1])) {
 344                 $decimals = mb_strlen(rtrim($matches[1][0], 0));
 345             }
 346         }
 347 
 348         return number_format($formater->parse($number, \NumberFormatter::TYPE_DOUBLE), $decimals, '.', ($showThousands ? ',' : ''));
 349     }

 420      * @return string
 421      */
 422     public static function maskString($val, $mask)
 423     {
 424         if (empty($val)) {
 425             return $val;
 426         }
 427         $maskared = '';
 428         $k = 0;
 429         if (is_numeric($val)) {

 432         for ($i = 0; $i <= mb_strlen($mask) - 1; $i++) {

 537      *
 538      * @return int
 539      *
 540      */
 541     public static function modulo11($n, $factor = 2, $base = 9, $x10 = 0, $resto10 = 0)
 542     {
 543         $sum = 0;
 544         for ($i = mb_strlen($n); $i > 0; $i--) {
 545             $sum += ((int) mb_substr($n, $i - 1, 1))*$factor;
 546             if ($factor == $base) {
 547                 $factor = 1;
 548             }
 549             $factor++;
 550         }
 551 
 552         if ($x10 == 0) {
 553             $sum *= 10;
 554             $digito = $sum%11;
 555             if ($digito == 10) {
 556                 $digito = $resto10;
 557             }
 558             return $digito;
 559         }
 560         return $sum%11;
 561     }

 563     /**
 564      * @param $n
 565      *
 566      * @return int
 567      */
 568     public static function modulo10($n)
 569     {
 570         $chars = array_reverse(str_split($n, 1));
 571         $odd = array_intersect_key($chars, array_fill_keys(range(1, count($chars), 2), null));
 572         $even = array_intersect_key($chars, array_fill_keys(range(0, count($chars), 2), null));
 573         $even = array_map(
 574             function ($n) {
 575                 return ($n >= 5) ? 2*$n - 9 : 2*$n;
 576             }, $even
 577         );
 578         $total = array_sum($odd) + array_sum($even);
 579         return ((floor($total/10) + 1)*10 - $total)%10;
 580     }
